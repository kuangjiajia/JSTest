/**
 * @constructor()
 * @param {string} title - 书本的标题
 * @param {string} author - 书本的作者
 */
 
 function Book(title,author) {
     this.title = title 
     this.author = author
 }

 Book.prototype = {
     /**
      * 获取书本的标题
      * @returns {string|*}
      */
     getTitle: function() {
         return this.title 
     },
     /**
      * 获取作者的姓名
      * @returns {string|*}
      */
     getAuthor: function() {
         return this.author
     },
     /**
      * 设置书本的页数
      * @param pageNum {number} 页数
      */
     setPageNum: function(pageNum) {
         this.pageNum = pageNum
     }
 }
